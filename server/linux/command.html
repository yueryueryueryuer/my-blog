# 常用命令

# 查找大文件

find / -type f -size +1000M -print0 | xargs -0 du -h | sort -nr

# 查看端口占用

netstat -tunlp | grep 端口号

lsof -i:端口号

# base64 加密解密

  • 加密
echo 'abc' | base64
  • 解密
echo 'YWJjCg==' | base64 -d
